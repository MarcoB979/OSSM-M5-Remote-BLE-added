#pragma once
#include <Arduino.h>
#include <esp_err.h>

// Simple, minimal ESP-NOW comms API.
// - Small, self-contained: init/deinit, broadcast/unicast send, peer management.
// - Receive callback is provided so user code can implement pairing logic.

extern "C" {

typedef struct {
  float esp_speed;
  float esp_depth;
  float esp_stroke;
  float esp_sensation;
  float esp_pattern;
  bool esp_rstate;
  bool esp_connected;
  bool esp_heartbeat;
  int esp_command;
  float esp_value;
  int esp_target;
  int esp_sender;
} espnow_message_t;

using espnow_recv_cb_t = void (*)(const uint8_t *mac, const espnow_message_t *msg, int len);

// Initialize ESP-NOW on `channel`. Returns true on success.
bool EspNowComms_init(int channel);
// Deinit ESP-NOW (optional)
void EspNowComms_deinit();

// Register a receive callback. Called from esp-now recv context.
void EspNowComms_registerRecvCb(espnow_recv_cb_t cb);

#define ESP_NOW_CHANNEL 1

// Send helpers
esp_err_t EspNowComms_sendBroadcast(const espnow_message_t *msg, size_t len);
esp_err_t EspNowComms_sendUnicast(const uint8_t *mac, const espnow_message_t *msg, size_t len);

// Peer helpers
bool EspNowComms_isPeer(const uint8_t *mac);
esp_err_t EspNowComms_addPeer(const uint8_t *mac);
esp_err_t EspNowComms_removePeer(const uint8_t *mac);

// High-level helper: send a command/value to a logical target (OSSM, EJECT, FIST)
// Routes via known peer MAC if available, otherwise broadcasts.
esp_err_t EspNowComms_sendToTarget(int target, int command, float value);

// Start/stop automatic pairing heartbeats (init starts heartbeats automatically).
bool EspNowComms_startHeartbeats(uint32_t intervalMs = 5000);
void EspNowComms_stopHeartbeats();

} // extern "C"
