// Legacy globals expected by other modules
bool ossm_espnow_connected = false;  // visible to other modules
// UI flags (set from recv context, processed by EspNowProcessPendingUiUpdates)
static volatile bool esp_now_ui_pairing_complete = false;
static volatile bool esp_now_ui_limits_ready = false;
static volatile bool esp_now_ui_homing = false;

extern bool Ossm_paired; // defined in main_helper.cpp

static constexpr uint8_t BROADCAST_ADDR[6] = {0xFF,0xFF,0xFF,0xFF,0xFF,0xFF};
static int s_channel = 1;
static espnow_recv_cb_t s_recv_cb = nullptr;
static TaskHandle_t s_heartbeat_task = nullptr;

// Symbol used by other modules as the canonical broadcast address for discovery heartbeats.
// DO NOT MODIFY this array at runtime — use only for sending discovery heartbeats.
static uint8_t DEFAULT_Address[6] = {0xFF,0xFF,0xFF,0xFF,0xFF,0xFF};
static uint8_t s_eject_addr[6] = {0xFF,0xFF,0xFF,0xFF,0xFF,0xFF};
static uint8_t s_fist_addr[6]  = {0xFF,0xFF,0xFF,0xFF,0xFF,0xFF};
static uint8_t s_ossm_addr[6]  = {0xFF,0xFF,0xFF,0xFF,0xFF,0xFF};
static volatile bool s_ossm_paired = false;

static inline bool isBroadcast(const uint8_t *mac) {
  return memcmp(mac, BROADCAST_ADDR, 6) == 0;
}

static void onDataRecv(const uint8_t *mac, const uint8_t *incomingData, int len) {
  // Copy safely into our message struct (zero-init so legacy short frames are safe)
  espnow_message_t msg;
  memset(&msg, 0, sizeof(msg));
  const int copyLen = (len < (int)sizeof(msg)) ? len : (int)sizeof(msg);
  if (copyLen > 0) memcpy(&msg, incomingData, copyLen);

  // Basic debug log
  Serial.printf("espnow RX from %02X:%02X:%02X:%02X:%02X:%02X len=%d cmd=%d sender=%d target=%d hb=%d\n",
                mac[0], mac[1], mac[2], mac[3], mac[4], mac[5], len,
                msg.esp_command, msg.esp_sender, msg.esp_target, msg.esp_heartbeat ? 1 : 0);

  // Update known address maps based on esp_sender
  if (msg.esp_sender == EJECT_ID) {
    memcpy(s_eject_addr, mac, 6);
    EspNowComms_addPeer(mac);
  } else if (msg.esp_sender == FIST_ID) {
    memcpy(s_fist_addr, mac, 6);
    EspNowComms_addPeer(mac);
  } else if (msg.esp_sender == OSSM_ID || (msg.esp_sender == 0 && msg.esp_target == M5_ID && msg.esp_speed > 0 && msg.esp_depth > 0)) {
    // OSSM or legacy OSSM-like message: consider pairing
    memcpy(s_ossm_addr, mac, 6);
    s_ossm_paired = true;
    EspNowComms_addPeer(mac);
  }
  // Let addons handle frames first (they may consume the frame)
  if (EjectHandleIncomingEspNowFrame(mac, msg.esp_target, msg.esp_sender, msg.esp_command, msg.esp_value, msg.esp_heartbeat)) {
    return;
  }
  if (FistITHandleIncomingEspNowFrame(mac, msg.esp_target, msg.esp_sender, msg.esp_command, msg.esp_value, msg.esp_heartbeat)) {
    return;
  }

  // Pairing detection + security checks similar to old communications module
  auto isOssmMessage = [&](const uint8_t *macAddr, const espnow_message_t &m)->bool {
    bool senderIsOssm = (m.esp_sender == 0 || m.esp_sender == OSSM_ID);
    if (!Ossm_paired) {
      return (m.esp_target == M5_ID);
    }
    return (memcmp(macAddr, s_ossm_addr, 6) == 0);
  };

  // If this looks like a pairing frame for us and we're not paired yet, handle pairing
  if (msg.esp_target == M5_ID && !Ossm_paired && isOssmMessage(mac, msg)) {
    const bool explicitOssmSender = (msg.esp_sender == OSSM_ID);
    const bool legacyUnspecifiedSender = (msg.esp_sender == 0);
    const bool hasValidOssmLimits = (msg.esp_speed > 0.0f && msg.esp_depth > 0.0f);
    if (!(explicitOssmSender || (legacyUnspecifiedSender && hasValidOssmLimits))) {
      LogDebugPrioFormatted("ESP-NOW: Ignore pre-pair frame (sender=%d target=%d speed=%.1f depth=%.1f cmd=%d)\n",
                            msg.esp_sender, msg.esp_target, msg.esp_speed, msg.esp_depth, msg.esp_command);
      return;
    }

    // Save OSSM address and try to add peer
    memcpy(s_ossm_addr, mac, 6);
    EspNowComms_addPeer(mac);
    // Mirror pairing flag used elsewhere
    Ossm_paired = true;
    esp_now_ui_pairing_complete = true;

    // Update limits shared globals
    if (msg.esp_speed > 600.0f) speedlimit = 600.0f; else speedlimit = msg.esp_speed;
    maxdepthinmm = msg.esp_depth;

    // Send a confirm heartbeat/control back to OSSM
    espnow_message_t outgoing = {};
    outgoing.esp_target = OSSM_ID;
    outgoing.esp_sender = M5_ID;
    esp_err_t result = EspNowComms_sendUnicast(s_ossm_addr, &outgoing, sizeof(outgoing));
    LogDebugFormatted("EspNowComms: pairing confirm send result=%d\n", (int)result);

    if (result == ESP_OK) {
      Ossm_paired = true;
      if (hasValidOssmLimits) {
        ossm_espnow_connected = true;
        LogDebugPrio("ESP_NOW: OSSM confirmed connected");
        esp_now_ui_limits_ready = true;
      } else {
        esp_now_ui_homing = true;
      }
    }
  }

  // SECURITY: Verify incoming message is from paired OSSM, not an addon
  if (!isOssmMessage(mac, msg)) {
    LogDebugPrioFormatted(
      "ESP-NOW: Rejecting frame (from=%02X:%02X:%02X:%02X:%02X:%02X sender=%d target=%d paired=%d expected=%02X:%02X:%02X:%02X:%02X:%02X)\n",
      mac[0], mac[1], mac[2], mac[3], mac[4], mac[5],
      msg.esp_sender,
      msg.esp_target,
      Ossm_paired ? 1 : 0,
      s_ossm_addr[0], s_ossm_addr[1], s_ossm_addr[2], s_ossm_addr[3], s_ossm_addr[4], s_ossm_addr[5]);
    return;
  }

  // Deliver to registered higher-level callback
  if (s_recv_cb) {
    s_recv_cb(mac, &msg, len);
  }
}

static void onDataSent(const uint8_t *mac_addr, esp_now_send_status_t status) {
  (void)mac_addr;
  Serial.printf("espnow TX status=%d\n", (int)status);
}

bool EspNowComms_init(int channel) {
  s_channel = channel;
  WiFi.mode(WIFI_STA);
  esp_wifi_set_channel(s_channel, WIFI_SECOND_CHAN_NONE);

  if (esp_now_init() != ESP_OK) {
    Serial.println("esp_now_init failed");
    return false;
  }

  esp_now_register_recv_cb(onDataRecv);
  esp_now_register_send_cb(onDataSent);
  Serial.printf("EspNowComms: initialized on channel %d\n", s_channel);
  // Start default heartbeat task (broadcast discovery)
  EspNowComms_startHeartbeats(5000);
  return true;
}

void EspNowComms_deinit() {
  esp_now_unregister_recv_cb();
  esp_now_unregister_send_cb();
  esp_now_deinit();
}

void EspNowComms_registerRecvCb(espnow_recv_cb_t cb) {
  s_recv_cb = cb;
}

// Do not add broadcast address as a peer — broadcasts are sent without peers.
esp_err_t EspNowComms_addPeer(const uint8_t *mac) {
  if (isBroadcast(mac)) return ESP_OK;
  if (esp_now_is_peer_exist(mac)) return ESP_OK;
  esp_now_peer_info_t p = {};
  memcpy(p.peer_addr, mac, 6);
  p.channel = s_channel;
  p.encrypt = false;
  esp_err_t r = esp_now_add_peer(&p);
  if (r == ESP_ERR_ESPNOW_EXIST) {
    Serial.printf("EspNowComms: add_peer %02X:%02X:%02X:%02X:%02X:%02X -> exists (treated as OK)\n",
                  mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
    return ESP_OK;
  }
  if (r != ESP_OK) {
    // Try removing stale entry and re-add
    esp_err_t d = esp_now_del_peer(mac);
    Serial.printf("EspNowComms: del_peer (cleanup) %02X:%02X:%02X:%02X:%02X:%02X -> %s\n",
                  mac[0], mac[1], mac[2], mac[3], mac[4], mac[5], esp_err_to_name(d));
    esp_err_t r2 = esp_now_add_peer(&p);
    Serial.printf("EspNowComms: add_peer retry %02X:%02X:%02X:%02X:%02X:%02X -> %s\n",
                  mac[0], mac[1], mac[2], mac[3], mac[4], mac[5], esp_err_to_name(r2));
    return r2;
  }

  Serial.printf("EspNowComms: add_peer %02X:%02X:%02X:%02X:%02X:%02X -> %s\n",
                mac[0], mac[1], mac[2], mac[3], mac[4], mac[5], esp_err_to_name(r));
  return r;
}

esp_err_t EspNowComms_removePeer(const uint8_t *mac) {
  if (isBroadcast(mac)) return ESP_OK;
  if (!esp_now_is_peer_exist(mac)) return ESP_ERR_NOT_FOUND;
  esp_err_t r = esp_now_del_peer(mac);
  Serial.printf("EspNowComms: del_peer %02X:%02X:%02X:%02X:%02X:%02X -> %s\n",
                mac[0], mac[1], mac[2], mac[3], mac[4], mac[5], esp_err_to_name(r));
  return r;
}

bool EspNowComms_isPeer(const uint8_t *mac) {
  if (isBroadcast(mac)) return false;
  return esp_now_is_peer_exist(mac) != 0;
}

esp_err_t EspNowComms_sendBroadcast(const espnow_message_t *msg, size_t len) {
  return esp_now_send(BROADCAST_ADDR, reinterpret_cast<const uint8_t *>(msg), len);
}

esp_err_t EspNowComms_sendUnicast(const uint8_t *mac, const espnow_message_t *msg, size_t len) {
  if (isBroadcast(mac)) return EspNowComms_sendBroadcast(msg, len);
  // Ensure peer exists with correct channel before unicast
  if (!EspNowComms_isPeer(mac)) {
    esp_err_t r = EspNowComms_addPeer(mac);
    if (r != ESP_OK) return r;
    // small delay can help on some devices; optional
    delay(5);
  }
  return esp_now_send(mac, reinterpret_cast<const uint8_t *>(msg), len);
}

// Heartbeat task: broadcast discovery heartbeats when not paired, unicast to OSSM when paired
static void heartbeatTask(void *pv) {
  uint32_t intervalMs = (uint32_t)pv;
  if (intervalMs == 0) intervalMs = 5000;
  espnow_message_t hb = {};
  hb.esp_command = HEARTBEAT;
  hb.esp_heartbeat = true;
  hb.esp_target = OSSM_ID;
  hb.esp_sender = M5_ID;

  for (;;) {
    if (!s_ossm_paired) {
      // Use canonical DEFAULT_Address for discovery heartbeats so other modules see consistent broadcast usage
      esp_err_t r = esp_now_send(DEFAULT_Address, reinterpret_cast<const uint8_t *>(&hb), sizeof(hb));
      if (r == ESP_ERR_ESPNOW_NOT_FOUND) {
        // Some ESP-IDF versions require a peer entry even for broadcast; add temporary peer and retry
        Serial.println("EspNowComms: DEFAULT_Address not found, adding temporary peer and retrying");
        esp_now_peer_info_t p = {};
        memcpy(p.peer_addr, DEFAULT_Address, 6);
        p.channel = s_channel;
        p.encrypt = false;
        esp_err_t ar = esp_now_add_peer(&p);
        Serial.printf("EspNowComms: add temp peer DEFAULT_Address -> %s\n", esp_err_to_name(ar));
        if (ar == ESP_OK) {
          esp_err_t r2 = esp_now_send(DEFAULT_Address, reinterpret_cast<const uint8_t *>(&hb), sizeof(hb));
          Serial.printf("EspNowComms: HB TX broadcast retry (DEFAULT_Address) result=%s\n", esp_err_to_name(r2));
          esp_err_t dr = esp_now_del_peer(DEFAULT_Address);
          Serial.printf("EspNowComms: del temp peer DEFAULT_Address -> %s\n", esp_err_to_name(dr));
          r = r2;
        }
      }
      Serial.printf("EspNowComms: HB TX broadcast (DEFAULT_Address) result=%s\n", esp_err_to_name(r));
    } else {
      esp_err_t r = EspNowComms_sendUnicast(s_ossm_addr, &hb, sizeof(hb));
      Serial.printf("EspNowComms: HB TX unicast to %02X:%02X:%02X:%02X:%02X:%02X result=%s\n",
                    s_ossm_addr[0], s_ossm_addr[1], s_ossm_addr[2], s_ossm_addr[3], s_ossm_addr[4], s_ossm_addr[5], esp_err_to_name(r));
    }
    vTaskDelay(pdMS_TO_TICKS(intervalMs));
  }
}

bool EspNowComms_startHeartbeats(uint32_t intervalMs) {
  if (s_heartbeat_task != nullptr) return true;
  // Pass interval in ms via pv parameter cast
  BaseType_t ok = xTaskCreatePinnedToCore(heartbeatTask, "espnow_heartbeat", 2048, (void *)intervalMs, 5, &s_heartbeat_task, 0);
  return ok == pdPASS;
}

void EspNowComms_stopHeartbeats() {
  if (s_heartbeat_task) {
    vTaskDelete(s_heartbeat_task);
    s_heartbeat_task = nullptr;
  }
}

// Route a logical target -> send correct struct_message to known MAC or broadcast
esp_err_t EspNowComms_sendToTarget(int target, int command, float value) {
  espnow_message_t msg = {};
  msg.esp_command = command;
  msg.esp_value = value;
  msg.esp_target = target;
  msg.esp_sender = M5_ID;

  if (target == OSSM_ID) {
    if (!s_ossm_paired) return EspNowComms_sendBroadcast(&msg, sizeof(msg));
    return EspNowComms_sendUnicast(s_ossm_addr, &msg, sizeof(msg));
  }
  if (target == EJECT_ID) {
    bool known = (memcmp(s_eject_addr, BROADCAST_ADDR, 6) != 0);
    if (known) return EspNowComms_sendUnicast(s_eject_addr, &msg, sizeof(msg));
    return EspNowComms_sendBroadcast(&msg, sizeof(msg));
  }
  if (target == FIST_ID) {
    // Fist handled by its own module; fall through to broadcast
    return EspNowComms_sendBroadcast(&msg, sizeof(msg));
  }

  // Default: broadcast
  return EspNowComms_sendBroadcast(&msg, sizeof(msg));
}

void EspNowInitCommunication()
{
  // Ensure station mode and set explicit channel to match OSSM
  WiFi.mode(WIFI_STA);
  esp_wifi_set_channel(ESP_NOW_CHANNEL, WIFI_SECOND_CHAN_NONE);
  LogDebugFormatted("ESP-NOW channel set to %d\n", ESP_NOW_CHANNEL);

  if (!EspNowComms_init(ESP_NOW_CHANNEL)) {
    Serial.println("Error initializing ESP-NOW");
  } else {
    Serial.println("esp_now_init ok");
  }
}

void EspNowSendPairingHeartbeat()
{
    EspNowComms_addPeer(s_ossm_addr); // Ensure OSSM is in peer list for unicast (no-op if already added)
    // EspNowComms_addPeer(s_eject_addr); // EJECT handled by its own module
    // EspNowComms_addPeer(s_fist_addr);  // FIST handled by its own module
    EspNowComms_addPeer(DEFAULT_Address); // Ensure DEFAULT_Address is in peer list for broadcast (some ESP-IDF versions require this)

  // Debug: print currently-known peers and whether they're registered with esp_now
  Serial.println("EspNowComms: peer list before sending heartbeat to all:");
  Serial.printf(" - GENERAL addr %02X:%02X:%02X:%02X:%02X:%02X -> %s\n",
                DEFAULT_Address[0], DEFAULT_Address[1], DEFAULT_Address[2], DEFAULT_Address[3], DEFAULT_Address[4], DEFAULT_Address[5],
                (isBroadcast(DEFAULT_Address) ? "broadcast" : (esp_now_is_peer_exist(DEFAULT_Address) ? "peer" : "no-peer")));
  Serial.printf(" - OSSM addr %02X:%02X:%02X:%02X:%02X:%02X -> %s\n",
                s_ossm_addr[0], s_ossm_addr[1], s_ossm_addr[2], s_ossm_addr[3], s_ossm_addr[4], s_ossm_addr[5],
                (isBroadcast(s_ossm_addr) ? "broadcast" : (esp_now_is_peer_exist(s_ossm_addr) ? "peer" : "no-peer")));
  // Eject/Fist peer info omitted here; those addons manage their own reporting

  
  espnow_message_t hb = {};
  hb.esp_command = HEARTBEAT;
  hb.esp_heartbeat = true;
  hb.esp_target = OSSM_ID;
  hb.esp_sender = M5_ID;
  // Always broadcast discovery heartbeat using DEFAULT_Address (do not modify DEFAULT_Address elsewhere)
  esp_err_t r = esp_now_send(DEFAULT_Address, reinterpret_cast<const uint8_t *>(&hb), sizeof(hb));
  if (r == ESP_ERR_ESPNOW_NOT_FOUND) {
    Serial.println("EspNowComms: DEFAULT_Address not found, adding temporary peer and retrying");
    esp_now_peer_info_t p = {};
    memcpy(p.peer_addr, DEFAULT_Address, 6);
    p.channel = s_channel;
    p.encrypt = false;
    esp_err_t ar = esp_now_add_peer(&p);
    Serial.printf("EspNowComms: add temp peer DEFAULT_Address -> %s\n", esp_err_to_name(ar));
    if (ar == ESP_OK) {
      esp_err_t r2 = esp_now_send(DEFAULT_Address, reinterpret_cast<const uint8_t *>(&hb), sizeof(hb));
      Serial.printf("EspNowComms: pairing heartbeat broadcast retry (DEFAULT_Address) result=%s\n", esp_err_to_name(r2));
      esp_err_t dr = esp_now_del_peer(DEFAULT_Address);
      Serial.printf("EspNowComms: del temp peer DEFAULT_Address -> %s\n", esp_err_to_name(dr));
      r = r2;
    }
  }
  Serial.printf("EspNowComms: pairing heartbeat broadcast (DEFAULT_Address) result=%s\n", esp_err_to_name(r));
  // Also send unicast confirm if we know OSSM address
  if (!isBroadcast(s_ossm_addr)) {
    EspNowComms_sendUnicast(s_ossm_addr, &hb, sizeof(hb));
  }
}

void EspNowWaitForPairingOrTimeout(uint32_t timeoutMs, uint32_t heartbeatIntervalMs)
{
  uint32_t waitStartMs = millis();
  uint32_t lastHeartbeatMs = 0;

  while (!Ossm_paired && (millis() - waitStartMs) < timeoutMs) {
    if ((millis() - lastHeartbeatMs) >= heartbeatIntervalMs) {
      EspNowSendPairingHeartbeat();
      lastHeartbeatMs = millis();
    }

    lv_task_handler();
    M5.update();
    vTaskDelay(pdMS_TO_TICKS(20));
  }
}

// C-callable wrapper used by UI C code
void EspNowWaitForPairingOrTimeout_c(uint32_t timeoutMs, uint32_t heartbeatIntervalMs)
{
  EspNowWaitForPairingOrTimeout(timeoutMs, heartbeatIntervalMs);
}

bool EspNowSendControlCommand(int command, float value, int target)
{
  // If target is OSSM, require pairing or an active espnow connection
  if (target == OSSM_ID && !Ossm_paired && !ossm_espnow_connected) {
    LogDebugFormatted("TX ESP blocked: not paired and not connected cmd=%d val=%.2f target=%d\n", command, value, target);
    return false;
  }

  espnow_message_t msg = {};
  msg.esp_connected = true;
  msg.esp_command = command;
  msg.esp_value = value;
  msg.esp_target = target;
  msg.esp_sender = M5_ID;

  // Route to specific targets: OSSM preferred, otherwise use sendToTarget helper
  if (target == OSSM_ID) {
    Serial.printf("TX ESP send cmd=%d val=%.2f target=%d sender=%d to=%02X:%02X:%02X:%02X:%02X:%02X\n",
                  command, value, target, msg.esp_sender,
                  s_ossm_addr[0], s_ossm_addr[1], s_ossm_addr[2], s_ossm_addr[3], s_ossm_addr[4], s_ossm_addr[5]);
    esp_err_t result = EspNowComms_sendUnicast(s_ossm_addr, &msg, sizeof(msg));
    if (result == ESP_OK) {
      LogDebug("TX ESP send result=OK");
      return true;
    }
    LogDebugFormatted("TX ESP send result=FAIL err=%d (retrying once)\n", (int)result);
    vTaskDelay(pdMS_TO_TICKS(20));
    esp_err_t retry = EspNowComms_sendUnicast(s_ossm_addr, &msg, sizeof(msg));
    LogDebugFormatted("TX ESP retry result=%s err=%d\n", (retry == ESP_OK) ? "OK" : "FAIL", (int)retry);
    return (retry == ESP_OK);
  }

  esp_err_t r = EspNowComms_sendToTarget(target, command, value);
  return (r == ESP_OK);
}

// Process pending UI updates from ESP-NOW pairing/data in a thread-safe way
// Call this from the main task loop to safely update LVGL UI
void EspNowProcessPendingUiUpdates()
{
  // Check if pairing just completed
  if (esp_now_ui_pairing_complete) {
    esp_now_ui_pairing_complete = false;
    lv_label_set_text(ui_Welcome, T_ESPCONNECTED);
  }

  // Check if limits are ready and we should show home screen
  if (esp_now_ui_limits_ready) {
    if (isStartScreenMinTimeElapsed()) {
      esp_now_ui_limits_ready = false;
      lv_label_set_text(ui_connect, "WIFI");
      lv_scr_load_anim(ui_Home, LV_SCR_LOAD_ANIM_FADE_ON, 20, 0, false);
    }
  }

  // Check if homing is in progress
  if (esp_now_ui_homing) {
    esp_now_ui_homing = false;
    waiting_for_limits = true;
    lv_label_set_text(ui_Welcome, T_HOMING);
  }
}
